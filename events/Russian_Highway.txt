

namespace = russian_highway

#advance_a_stage
province_event = {
	id = russian_highway.1
	title = russian_highway.1.t
	picture = FARMING_eventPicture
	goto = root
	
	desc = {
		trigger = {
			has_province_modifier = russian_highway_stage_1
		}
		desc = russian_highway.1.da
	}
	desc = {
		trigger = {
			has_province_modifier = russian_highway_stage_2
		}
		desc = russian_highway.1.db
	}
	desc = {
		trigger = {
			NOT = {
				has_province_modifier = russian_highway_stage_1
				has_province_modifier = russian_highway_stage_2
			}
		}
		desc = russian_highway.1.dc
	}
	
	trigger = {
		has_province_modifier = russian_highway_under_construction
	}
	
	mean_time_to_happen = {
		days = 1500
		modifier = {
			factor = 0.75
			owner = { full_idea_group = economic_ideas }
		}
		modifier = {
			factor = 0.8
			owner = { full_idea_group = administrative_ideas }
		}
		modifier = {
			factor = 0.9
			development = 10
		}
		modifier = {
			factor = 0.9
			development = 15
		}
		modifier = {
			factor = 0.8
			development = 20
		}
		modifier = {
			factor = 0.7
			development = 30
		}
		modifier = {
			factor = 0.7
			is_prosperous = yes
		}
		modifier = {
			factor = 2
			devastation = 1
		}
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = { has_province_modifier = russian_highway_stage_1 }
				
				remove_province_modifier = russian_highway_stage_1
				add_permanent_province_modifier = {
					name = russian_highway_stage_2
					duration = -1
				}
			}
			else_if = {
				limit = { has_province_modifier = russian_highway_stage_2 }
				
				remove_province_modifier = russian_highway_under_construction
				remove_province_modifier = russian_highway_stage_2
				add_permanent_province_modifier = {
					name = russian_highway
					duration = -1
				}
			}
			else = {
				add_permanent_province_modifier = {
					name = russian_highway_stage_1
					duration = -1
				}
			}
		}
	}

	#Continue building
	option = {
		name = russian_highway.1.a
		ai_chance = { factor = 1 }
		
		trigger = { NOT = { has_province_modifier = russian_highway } }
		
		owner = { add_treasury = -30 }
	}
	
	#We don't have the funds for now
	option = {
		name = russian_highway.1.b
		ai_chance = { factor = 0 }
		
		trigger = { NOT = { has_province_modifier = russian_highway } }
				
		remove_province_modifier = russian_highway_under_construction
		add_permanent_province_modifier = {
			name = russian_highway_construction_stalled
			duration = -1
		}
		
		owner = { add_prestige = -5 }
	}
	
	#The road restored!
	option = {
		name = russian_highway.1.c
		ai_chance = { factor = 1 }
		
		trigger = { has_province_modifier = russian_highway }
		
		owner = { add_prestige = 2 }
	}
}
