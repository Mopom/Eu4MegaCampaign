country_event = {
	id = flavor_aez.1
	title = "flavor_aez.1.t"
	desc = "flavor_aez.1.d"
	picture = BANKRUPTCY_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		hidden_effect = {
			every_province = {
				limit = {
					is_capital_of = AEZ
					NOT = {
						province_id = 808
					}
				}
				change_province_name = "Manjam al-Fida"
				rename_capital = "Manjam al-Fida"
			}
		}
	}
	
	option = {
		name = "flavor_aez.1.a"
		MAL = {
			release_all_possible_countries = yes
			set_bankruptcy = ROOT
			release_all_subjects = yes
		}
	}
	
	option = {
		name = "flavor_aez.1.b"
		switch_tag = AEZ
		MAL = {
			release_all_possible_countries = yes
			set_bankruptcy = MAL
			release_all_subjects = yes
		}
		AEZ = {
		    country_event = { id = ideagroups.1 } #Swap Ideas
		}
	}
	
	after = {
		MAL = {
			every_owned_province = {
				add_unrest = 30
			}
		}
	}
}
