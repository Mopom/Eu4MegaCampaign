				rename_capital = "Nouveau-Bordeaux"
			}
			every_province = {

					AND = {

						is_capital_of = ROOT
					}

				change_province_name = "Lyeste"
				rename_capital = "La Lyon"
			}
		}
	}
	option = { #monarchist
		name = "monarch_colony.1.a"

		name = "monarch_colony.1.b"
		ai_chance = {factor = 25}
		break_alliance = FRA
