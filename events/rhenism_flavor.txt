#The Prophet
#1450
country_event = { 
	id = rhenism.1
	title = rhenism.1.t
	desc = rhenism.1.d
	picture = PRAYING_eventPicture

	is_triggered_only = yes

	option = { # Wise man
		name = rhenism.1.a
		add_country_modifier = {
			name = prophet_wise_man
			duration = -1
		}

		every_subject_country = {
			add_country_modifier = {
				name = prophet_wise_man
				duration = -1
			}
		}
	}

	option = { #Inspired, but not absolute
		name = rhenism.1.b
		add_country_modifier = {
			name = prophet_inspired
			duration = -1
		}

		every_subject_country = {
			add_country_modifier = {
				name = prophet_inspired
				duration = -1
			}
		}
	}

	option = { #The words of The Prophet reign supreme
		name = rhenism.1.c
		add_country_modifier = {
			name = prophet_supreme
			duration = -1
		}

		every_subject_country = {
			add_country_modifier = {
				name = prophet_supreme
				duration = -1
			}
		}
	}
}

country_event = { #Driver for above event
	id = rhenism.2

	hidden = yes
	fire_only_once = yes

	trigger = {
		AND = {
			religion = rhenism
			total_development = 300
			is_year = 1450
		}
	}

	mean_time_to_happen = {
		months = 60

		modifier = {
			factor = 0.5
			is_year = 1460
		}
	}

	option = {
		ai_chance = {factor = 1}
		every_country = {
			limit = {
				religion = rhenism
				is_subject = no
			}
			country_event = {
				id = rhenism.1
			}
		}
	}
}
