# Refusal of Local Court to Agree to Ordonances and Taxes
country_event = {
	id = gngng_events.2
	title = "gngng_events.2.t"
	desc = "gngng_events.2.d"
	picture = BAD_WITH_MONARCH_eventPicture

	is_triggered_only = yes
	
	trigger = {
		any_owned_province = {
			is_capital = no
			is_state = yes
			NOT = { has_province_modifier = "cnc_refused_to_pay_taxes" }
			NOT = { has_province_flag = had_local_courts_refused }
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					is_state = yes
					NOT = { has_province_modifier = "cnc_refused_to_pay_taxes" }
					NOT = { has_province_flag = had_local_courts_refused }
				}
				save_event_target_as = local_court_location
				area = {
					limit = {
						owned_by = ROOT
						is_capital = no
						NOT = { has_province_flag = local_courts_refuse_to_pay_tax }
					}
					set_province_flag = local_courts_refuse_to_pay_tax
				}
			}
		}
	}
	
	option = {		# 
		name = "gngng_events.2.a"
		event_target:local_court_location = {
			spawn_rebels = {
				type = particularist_rebels
				size = 1.5
			}
		}
		every_owned_province = {
			limit = {
				has_province_flag = local_courts_refuse_to_pay_tax
			}
			add_province_modifier = {
				name = "cnc_local_authority_asserted"
				duration = 1825
			}
			set_province_flag = had_local_courts_refused
			clr_province_flag = local_courts_refuse_to_pay_tax
		}
	}
	option = {		# 
		name = "gngng_events.2.b"
		every_owned_province = {
			limit = {
				has_province_flag = local_courts_refuse_to_pay_tax
			}
			add_province_modifier = {
				name = "cnc_refused_to_pay_taxes"
				duration = 3650
			}
			set_province_flag = had_local_courts_refused
			clr_province_flag = local_courts_refuse_to_pay_tax
		}
	}
}

# Growing Separatism
# Like, for instance, the Catalan revolt.
country_event = {
	id = gngng_events.3
	title = "gngng_events.3.t"
	desc = "gngng_events.3.d"
	picture = BORDER_TENSION_eventPicture


	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = cnc_dealt_with_separatists }
		num_of_cities = 15
		any_owned_province = {
			has_owner_accepted_culture = no
			NOT = { has_province_flag = separatist_province_set_here }
			any_core_country = {
				NOT = { tag = ROOT }
				primary_culture = PREV
				calc_true_if = {
					all_core_province = {
						owned_by = ROOT
						culture = PREV
					}
					amount = 5
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = cnc_dealt_with_separatists #Only once per game

				limit = {
					NOT = { has_province_flag = separatist_province_set_here }

						NOT = { tag = ROOT }
						primary_culture = PREV

							all_core_province = {

							}
							amount = 5
						}
					}
				}
				set_province_flag = separatist_province_set_here

					limit = {
						NOT = { tag = ROOT }
						primary_culture = PREV

								owned_by = ROOT

# Quarrelsome Nobles
country_event = {
	id = gngng_events.5
	title = "gngng_events.5.t"
	desc = "gngng_events.5.d"
	picture = SEIGE_eventPicture


	is_triggered_only = yes

	trigger = {
		any_owned_province = { #Only Once per Province
			NOT = { has_province_flag = cnc_quarrelsome_nobles }
			NOT = {
				any_neighbor_province = {
					owned_by = ROOT
					is_capital = yes
				}
			}
		}
		NOT = {
			calc_true_if = {
				all_owned_province = {
					has_province_flag = cnc_quarrelsome_nobles
				}
				amount = 2 #At most 2 Times during disaster
			}
		}
	}
	
	immediate = {
		hidden_effect = {	#Picking out suitable province to use:
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cnc_quarrelsome_nobles }
					NOT = {
						any_neighbor_province = {
							owned_by = ROOT
							is_capital = yes
						}
					}
				}
				save_event_target_as = quarrelsome_nobles_province
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cnc_quarrelsome_nobles }
					NOT = {
						any_neighbor_province = {
							owned_by = ROOT
							is_capital = yes
						}
					}
					OR = {
						owner = { has_estate = estate_nobles }
						has_province_modifier = domain_of_spouses_family
						has_province_modifier = angered_nobles
					}
				}
				save_event_target_as = quarrelsome_nobles_province
			}
		}
	}

	mean_time_to_happen = {
		days = 1
		
	}
	
	option = {
		name = "gngng_events.5.a" #
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0.2
				is_at_war = yes
			}
		}
		event_target:quarrelsome_nobles_province = {
			set_province_flag = cnc_quarrelsome_nobles
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		if = {
			limit = {
				num_of_cities = 30
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cnc_quarrelsome_nobles }
					is_capital = no
					is_in_capital_area = yes
				}
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
			}
		}
		if = {
			limit = {
				num_of_cities = 40
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cnc_quarrelsome_nobles }
					is_capital = no
					is_in_capital_area = yes
				}
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
			}
		}
		if = {
			limit = {
				num_of_cities = 60
			}
			random_owned_province = {
				limit = {
					NOT = { has_province_flag = cnc_quarrelsome_nobles }
					is_capital = no
					is_in_capital_area = yes
				}
				spawn_rebels = {
					type = noble_rebels
					size = 1
				}
			}
		}
	}
	option = {
		name = "gngng_events.5.b" # 
		ai_chance = { factor = 20 }
		add_legitimacy = -10
		add_devotion = -10
		add_horde_unity = -10
